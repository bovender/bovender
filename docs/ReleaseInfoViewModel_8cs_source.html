131  protected override void SendProcessFinishedMessage()
132  {
133  base.SendProcessFinishedMessage();
134  switch (Status)
135  {
136  case ReleaseInfoStatus.InfoAvailable:
137  ProcessMessageContent.WasSuccessful = true;
138  if (ReleaseVersion > CurrentVersion)
139  {
140  Logger.Info("SendProcessFinishedMessage: update available, {0} > {1}",
141  ReleaseVersion.ToString(), CurrentVersion.ToString());
142  UpdateAvailableMessage.Send(ProcessMessageContent);
143  }
144  else
145  {
146  Logger.Info("SendProcessFinishedMessage: no update available, {0} <= {1}",
147  ReleaseVersion.ToString(), CurrentVersion.ToString());
148  NoUpdateAvailableMessage.Send(ProcessMessageContent);
149  }
150  break;
151  case ReleaseInfoStatus.FailureToFetch:
152  case ReleaseInfoStatus.FailureToParse:
153  Logger.Warn("SendProcessFinishedMessage: Exception occurred!");
154  Logger.Warn(ReleaseInfo.Exception);
155  ProcessMessageContent.Exception = ReleaseInfo.Exception;
156  ProcessMessageContent.WasSuccessful = false;
157  ExceptionMessage.Send(ProcessMessageContent);
158  break;
159  default:
160  break;
161  }
162  }
