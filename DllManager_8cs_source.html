73  public void LoadDll(string dllName)
74  {
75  // Check if the DLL exists
76  string dllPath = CompletePath(dllName);
77  CheckFilePresent(dllPath);
78 
79  // Attempt to load the DLL
80  IntPtr handle = LoadLibrary(dllPath);
81  if (handle == IntPtr.Zero)
82  {
83  // Strip the leading directories from the path info (they may contain
84  // sensitive information about where exactly a user has installed files).
85  string[] dirs = Path.GetDirectoryName(dllPath).Split(Path.DirectorySeparatorChar);
86  string gracefulPath = dllName;
87  int n = dirs.Length;
88  if (n > 0) gracefulPath = Path.Combine(dirs[n - 1], gracefulPath);
89  if (n > 1) gracefulPath = Path.Combine(dirs[n - 2], gracefulPath);
90  if (n > 2) gracefulPath = Path.Combine("...", gracefulPath);
91  Win32Exception inner = new Win32Exception(Marshal.GetLastWin32Error());
92  throw new DllLoadingFailedException(
93  String.Format(
94  "LoadLibrary failed with code {0} on {1}",
95  Marshal.GetLastWin32Error(), gracefulPath
96  ),
97  inner
98  );
99  }
100 
101  // Register the DLL and its handle in the internal database
102  _loadedDlls.Add(dllName, handle);
103  }
