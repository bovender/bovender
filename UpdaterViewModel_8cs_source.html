407  void _updater_CheckForUpdateFinished(object sender, EventArgs e)
408  {
409  IsLocked = false;
410  Action action = new Action(() =>
411  {
412  // Set the 'IsIndeterminate' property of the process message content to false
413  // so that the progress bar stops its animation.
414  CheckProcessMessageContent.IsIndeterminate = false;
415  if (_updater.DownloadException == null)
416  {
417  if (_updater.IsUpdateAvailable)
418  {
419  if (_updater.IsAuthorized)
420  {
421  UpdateAvailableMessage.Send(
422  CheckProcessMessageContent,
423  (content) => InstallUpdateCommand.Execute(null));
424  }
425  else
426  {
427  UpdateAvailableButNotAuthorizedMessage.Send(
428  CheckProcessMessageContent,
429  (content) => CloseViewCommand.Execute(null));
430  }
431  }
432  else
433  {
434  NoUpdateAvailableMessage.Send(
435  CheckProcessMessageContent,
436  (content) => CloseViewCommand.Execute(null));
437  }
438  }
439  else
440  {
441  OnPropertyChanged("DownloadException");
442  NetworkFailureMessage.Send(
443  CheckProcessMessageContent,
444  (content) => CloseViewCommand.Execute(null));
445  }
446  });
447 
448  // Asynchronous operations that interact with GUI views must be invoked
449  // via the view's dispatcher. If there is no GUI view and hence no
450  // dispatcher, simply invoke the action itself (e.g., if the view
451  // is a non-GUI object).
452  if (ViewDispatcher != null)
453  {
454  ViewDispatcher.Invoke(action);
455  }
456  else {
457  action.Invoke();
458  }
459  }

491  void _updater_DownloadUpdateFinished(object sender, EventArgs e)
492  {
493  IsLocked = false;
494  // When the download process is finished, the Updater model will verify
495  // the downloaded file's Sha1. Since the view model exposes the
496  // updater's IsVerifiedDownload, we need to notify subscribers to the
497  // view model's INotifyPropertyChanged interface (inherited from
498  // ViewModelBase).
499  OnPropertyChanged("IsVerifiedDownload");
500  OnPropertyChanged("IsUpdatePending");
501 
502  if (_updater.DownloadException == null)
503  {
504  if (_updater.IsAuthorized)
505  {
506  if (_updater.IsVerifiedDownload)
507  {
508  UpdateInstallableMessage.Send(DownloadProcessMessageContent);
509  }
510  else
511  {
512  UpdateFailedVerificationMessage.Send(DownloadProcessMessageContent);
513  }
514  }
515  else
516  {
517  UpdateAvailableButNotAuthorizedMessage.Send(DownloadProcessMessageContent);
518  }
519  }
520  else
521  {
522  NetworkFailureMessage.Send(DownloadProcessMessageContent);
523  }
524  }
