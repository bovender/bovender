395  private void DoCheckForUpdate()
396  {

403  void _updater_CheckForUpdateFinished(object sender, EventArgs e)
404  {
405  Logger.Info("_updater_CheckForUpdateFinished");
406  IsLocked = false;
407  Action action = new Action(() =>
408  {
409  // Set the 'IsIndeterminate' property of the process message content to false
410  // so that the progress bar stops its animation.
411  CheckProcessMessageContent.IsIndeterminate = false;
412  if (_updater.DownloadException == null)
413  {
414  if (_updater.IsUpdateAvailable)
415  {
416  if (_updater.IsAuthorized)
417  {
418  Logger.Info("Update available, user authorized");
419  UpdateAvailableMessage.Send(
420  CheckProcessMessageContent,
421  (content) => InstallUpdateCommand.Execute(null));
422  }
423  else
424  {
425  Logger.Info("Update available, but user not authorized");
426  UpdateAvailableButNotAuthorizedMessage.Send(
427  CheckProcessMessageContent,
428  (content) => CloseViewCommand.Execute(null));
429  }
430  }
431  else
432  {
433  Logger.Info("No update available");
434  NoUpdateAvailableMessage.Send(
435  CheckProcessMessageContent,
436  (content) => CloseViewCommand.Execute(null));
437  }
438  }
439  else
440  {
441  Logger.Warn(_updater.DownloadException, "Download exception");
442  OnPropertyChanged("DownloadException");
443  NetworkFailureMessage.Send(
444  CheckProcessMessageContent,
445  (content) => CloseViewCommand.Execute(null));
446  }
447  });
448 
449  // Asynchronous operations that interact with GUI views must be invoked
450  // via the view's dispatcher. If there is no GUI view and hence no
451  // dispatcher, simply invoke the action itself (e.g., if the view
452  // is a non-GUI object).
453  if (ViewDispatcher != null)
454  {
455  ViewDispatcher.Invoke(action);
456  }
457  else {
458  action.Invoke();
459  }
460  }

493  void _updater_DownloadUpdateFinished(object sender, EventArgs e)
494  {
495  Logger.Info("_updater_DownloadUpdateFinished");
496  IsLocked = false;
497  // When the download process is finished, the Updater model will verify
498  // the downloaded file's Sha1. Since the view model exposes the
499  // updater's IsVerifiedDownload, we need to notify subscribers to the
500  // view model's INotifyPropertyChanged interface (inherited from
501  // ViewModelBase).
502  OnPropertyChanged("IsVerifiedDownload");
503  OnPropertyChanged("IsUpdatePending");
504 
505  if (_updater.DownloadException == null)
506  {
507  if (_updater.IsAuthorized)
508  {
509  if (_updater.IsVerifiedDownload)
510  {
511  Logger.Info("Download verified");
512  UpdateInstallableMessage.Send(DownloadProcessMessageContent);
513  }
514  else
515  {
516  Logger.Info("Failed verficiation");
517  UpdateFailedVerificationMessage.Send(DownloadProcessMessageContent);
518  }
519  }
520  else
521  {
522  Logger.Info("User not authorized");
523  UpdateAvailableButNotAuthorizedMessage.Send(DownloadProcessMessageContent);
524  }
525  }
526  else
527  {
528  Logger.Warn(DownloadProcessMessageContent.Exception, "Network failure");
529  NetworkFailureMessage.Send(DownloadProcessMessageContent);
530  }
531  }
