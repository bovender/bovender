78  protected static T FromFileOrDefault<T>(string yamlFile)
79  where T: UserSettingsBase, new()
80  {
81  T optionsStore = null;
82  if (File.Exists(yamlFile))
83  {
84  try
85  {
86  using (FileStream fs = File.Open(yamlFile, FileMode.Open, FileAccess.Read))
87  {
88  StreamReader sr = new StreamReader(fs);
89  Deserializer des = new Deserializer(ignoreUnmatched: true);
90  optionsStore = des.Deserialize<T>(sr);
91  if (optionsStore != null)
92  {
93  optionsStore.WasFromFile = true;
94  }
95  }
96  Logger.Info("Loaded user settings from file '{0}'", yamlFile);
97  }
98  catch (IOException e)
99  {
100  optionsStore = CreateDefaultOnException<T>(e);
101  }
102  catch (YamlException e)
103  {
104  optionsStore = CreateDefaultOnException<T>(e);
105  }
106  }
107  if (optionsStore == null)
108  {
109  Logger.Info("Creating user settings object from scratch");
110  optionsStore = new T();
111  }
112  return optionsStore;
113  }

232  public void Save()
233  {

237  Directory.CreateDirectory(System.IO.Path.GetDirectoryName(fn));
238  using (FileStream fs = File.Open(fn, FileMode.Create, FileAccess.Write))
239  {
240  StreamWriter sw = new StreamWriter(fs, Encoding.UTF8);
241  WriteYamlHeader(sw);
242  Serializer ser = new Serializer();
243  ser.Serialize(sw, this);
244  sw.Flush();
245  }
246  Exception = null;
247  Logger.Info("Loaded user settings from file '{0}'", fn);
248  }
249  catch (IOException e)
250  {
251  Logger.Warn("Could not save user settings", e);
252  Exception = e;
253  }
