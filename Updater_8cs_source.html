167  public void DownloadUpdate()
168  {
169  _destinationFileName = BuildDestinationFileName();
170 
171  /* Check if the file exists already. If the Sha1 is identical,
172  * do not download it again. If the Sha1 is different, it is a file
173  * with the same name, but different content (broken download?).
174  */
175  if (File.Exists(_destinationFileName) &&
176  FileHelpers.Sha1Hash(_destinationFileName) == UpdateSha1)
177  {
178  // Bypass the download and signal that the file is present
179  IsVerifiedDownload = true;
180  OnDownloadUpdateFinished();
181  }
182  else
183  {
184  _client = new WebClient();
185  _client.DownloadProgressChanged += _client_DownloadProgressChanged;
186  _client.DownloadFileCompleted += _client_DownloadFileCompleted;
187  _client.DownloadFileAsync(DownloadUri, _destinationFileName);

201  public void InstallUpdate()
202  {
203  if (!IsUpdateAvailable) return;
204 
205  // As a security measure, compute the SHA1 again so we know it's current.
206  IsVerifiedDownload = FileHelpers.Sha1Hash(_destinationFileName) == UpdateSha1;
207 
208  if (IsVerifiedDownload)
209  {
210  DoInstallUpdate();
211  }
212  else
213  {
214  throw new DownloadCorruptException("The Sha1 checksum of the file on disk is unexpected.");
215  }

371  void VersionInfoClient_DownloadStringCompleted(object sender, DownloadStringCompletedEventArgs e)
372  {
373  if (!e.Cancelled) {
374  if (e.Error == null )
375  {
376  StringReader r = new StringReader(e.Result);
377  NewVersion = new SemanticVersion(r.ReadLine());
378  string rawUri = r.ReadLine();
379  // If the raw URI contains the placeholder $VERSION, replace
380  // it with the new version.
381  DownloadUri = new Uri(rawUri.Replace("$VERSION", NewVersion.ToString()));
382  // Use only the first word of the line as Sha1 sum
383  // to make it compatible with the output of `sha1sum`
384  UpdateSha1 = r.ReadLine().Split(' ')[0];
385  Multiline multi = new Multiline(r.ReadToEnd(), true);
386  UpdateSummary = multi.Text;
387  IsUpdateAvailable = NewVersion > GetCurrentVersion();
388  }
389  else
390  {
391  DownloadException = e.Error;
392  IsUpdateAvailable = false;
393  }
394  OnCheckForUpdateFinished();
395  }
396  }
