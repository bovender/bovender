369  void VersionInfoClient_DownloadStringCompleted(object sender, DownloadStringCompletedEventArgs e)
370  {
371  if (!e.Cancelled) {
372  if (e.Error == null )
373  {
374  StringReader r = new StringReader(e.Result);
375  NewVersion = new SemanticVersion(r.ReadLine());
376  string rawUri = r.ReadLine();
377  // If the raw URI contains the placeholder $VERSION, replace
378  // it with the new version.
379  DownloadUri = new Uri(rawUri.Replace("$VERSION", NewVersion.ToString()));
380  // Use only the first word of the line as Sha1 sum
381  // to make it compatible with the output of `sha1sum`
382  UpdateChecksum = r.ReadLine().Trim().Split(' ')[0];
383  Multiline multi = new Multiline(r.ReadToEnd(), true);
384  UpdateSummary = multi.Text;
385  IsUpdateAvailable = NewVersion > GetCurrentVersion();
386  }
387  else
388  {
389  DownloadException = e.Error;
390  IsUpdateAvailable = false;
391  }
392  OnCheckForUpdateFinished();
393  }

401  bool VerifyDownload()
402  {
403  switch (UpdateChecksum.Length)
404  {
405  case 40:
406  IsVerifiedDownload = FileHelpers.Sha1Hash(_destinationFileName) == UpdateChecksum;
407  break;
408  case 64:
409  IsVerifiedDownload = FileHelpers.Sha256Hash(_destinationFileName) == UpdateChecksum;
410  break;
411  default:
412  IsVerifiedDownload = false;
413  break;
414  }
415  return IsVerifiedDownload;
